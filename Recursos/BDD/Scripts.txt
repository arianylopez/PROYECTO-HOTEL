SCRIPTS BASE DE DATOS - SUPABASE
TABLAS
CREATE TABLE TipoHabitacion (
    tipo_habitacion_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT,
    capacidad INT NOT NULL,
    precio_base DECIMAL(10, 2) NOT NULL,
    amenidades TEXT[] 
);

CREATE TABLE Habitacion (
    habitacion_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    numero_habitacion VARCHAR(20) NOT NULL UNIQUE,
    piso INT NOT NULL,
    tipo_habitacion_id UUID NOT NULL REFERENCES TipoHabitacion(tipo_habitacion_id),
    estado VARCHAR(50) DEFAULT 'Disponible' CHECK (estado IN ('Disponible', 'Ocupada'))
);

CREATE TABLE Huesped (
    huesped_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    tipo_documento VARCHAR(20) NOT NULL,
    documento_identidad VARCHAR(50) NOT NULL UNIQUE,
    telefono VARCHAR(50),
    correo VARCHAR(100),
    direccion TEXT,
    nacionalidad VARCHAR(50),
    fecha_registro DATE DEFAULT CURRENT_DATE
);

CREATE TABLE PoliticaCancelacion (
    politica_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    dias_limite_sin_mora INT NOT NULL,
    porcentaje_penalidad DECIMAL(3, 2) NOT NULL, 
    descripcion TEXT,
    activa BOOLEAN DEFAULT TRUE
);

CREATE TABLE Estadia (
    estadia_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    huesped_titular_id UUID NOT NULL REFERENCES Huesped(huesped_id),
    habitacion_id UUID NOT NULL REFERENCES Habitacion(habitacion_id),
    fecha_ingreso DATE NOT NULL,
    fecha_salida DATE NOT NULL,
    cantidad_personas INT NOT NULL,
    precio_aplicado DECIMAL(10, 2) NOT NULL,
    estado VARCHAR(50) DEFAULT 'Reservada' CHECK (estado IN ('Reservada', 'En curso', 'Finalizada', 'Cancelada')),
    observaciones TEXT,
    fecha_hora_checkin TIMESTAMP,
    fecha_hora_checkout TIMESTAMP,
    mora DECIMAL(10, 2) DEFAULT 0.00,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE EstadiaHuesped (
    estadia_acompañante_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    estadia_id UUID NOT NULL REFERENCES Estadia(estadia_id) ON DELETE CASCADE,
    huesped_id UUID NOT NULL REFERENCES Huesped(huesped_id) ON DELETE CASCADE
);

CREATE TABLE AreaServicio (
    area_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre_area VARCHAR(100) NOT NULL, 
    descripcion TEXT,
    horario_atencion VARCHAR(100),
    ubicacion VARCHAR(150)
);

CREATE TABLE Empleado (
    empleado_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre_empleado VARCHAR(100) NOT NULL,
    cargo VARCHAR(100) NOT NULL, 
    telefono VARCHAR(50),
    email VARCHAR(100)
);

CREATE TABLE AsignacionArea (
    asignacion_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    area_id UUID NOT NULL REFERENCES AreaServicio(area_id) ON DELETE CASCADE,
    empleado_id UUID NOT NULL REFERENCES Empleado(empleado_id) ON DELETE CASCADE,
    es_encargado BOOLEAN DEFAULT false, 
    UNIQUE(area_id, empleado_id) 
);
